DELAY 750

STRINGLN Welcome to duckyScript 3 beta test! Here's a quick demo.
ENTER

STRINGLN Variables and operations:
VAR $spam = 10
$spam = 10 * 4 + 2
STRINGLN 	Spam is: $spam
ENTER

//---------------
$spam = 0
$_RANDOM_MIN = 1
$_RANDOM_MAX = 10

STRINGLN Conditions and loops:

WHILE $spam < 4
    VAR $this_number = $_RANDOM_INT
    STRING 	Random number: $this_number

    IF $this_number > 5 THEN
        STRINGLN , it's more than 5!
    ELSE IF $this_number == 5 THEN
        STRINGLN , exactly 5!
    ELSE
        STRINGLN , it's less than 5!
    END_IF

    $spam = $spam + 1
END_WHILE
ENTER

//----------------
STRINGLN Printing to OLED screen
STRINGLN Look at the screen!

WHILE 1
    OLED_CLEAR
    OLED_CURSOR 20 0
    OLED_PRINT Hello World!
    OLED_CURSOR 20 12
    OLED_PRINT Press a key!
    OLED_CURSOR 10 52
    OLED_PRINT Press + to exit

    VAR $this_key = $_READKEY

    IF $this_key != 0 THEN
        OLED_CURSOR 10 31
        OLED_PRINT I pressed key $this_key!
    END_IF

    IF $this_key == 17 THEN
        BREAK
    END_IF

    OLED_UPDATE
    DELAY 50
END_WHILE



WHILE 1
	OLED_CLEAR
	OLED_CURSOR 20 20
	OLED_PRINT Continue demo?

	OLED_CURSOR 25 32
	OLED_PRINT - No + Yes
	OLED_UPDATE

	VAR $this_key = $_READKEY

	IF $this_key == 16 || $this_key == 17 THEN
        BREAK
    END_IF

	DELAY 50
END_WHILE

OLED_CLEAR

IF $this_key == 16 THEN
